LinQ for TypeScript

Install

npm install linqts

Usage

import {List} from 'linqts';

let arr = new List<number>([1,2,3,4,5])
    .Where(x => x > 3)
    .Select(y => y * 2)
    .ToArray(); // > [8, 10]

let query =
    people.Join(pets,
                person => person,
                pet => pet.Owner,
                (person, pet) =>
                    ({ OwnerName: person.Name, Pet: pet.Name }));
